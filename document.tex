\section*{連立微分方程式の数値解}

\eqn{
    \begin{cases}
        \dd{x}{t} &= - \sigma x + \sigma y \\
        \dd{y}{t} &= rx - y - xz \\
        \dd{z}{t} &= -bz + xy
    \end{cases}
}
\eqn{\sigma = 10,~ r = 28,~ b = \frac{8}{3}}
\eqn{x(0) = 0,~ y(0) = 1.1,~ z(0) = 0}
